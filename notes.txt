# Clean and rebuild
dotnet clean
dotnet restore
dotnet build

# If build succeeds, run tests
dotnet test
dotnet build 2>&1 | Select-String "error"


# Navigate to your Infrastructure project
cd InsureX.Infrastructure

# Add a new migration
dotnet ef migrations add AddPartnerDateFields --context ApplicationDbContext

# Update database
dotnet ef database update --context ApplicationDbContext

# List all migrations
dotnet ef migrations list --context ApplicationDbContext

# Script migrations (for production deployment)
dotnet ef migrations script --context ApplicationDbContext -o ./Migrations/Script.sql

# Remove last migration (if not applied)
dotnet ef migrations remove --context ApplicationDbContext

# Create a console app for seeding
dotnet new console -n InsureX.SeedTool
cd InsureX.SeedTool
dotnet add reference ../InsureX.Infrastructure
dotnet add reference ../InsureX.Domain
dotnet add package Microsoft.EntityFrameworkCore.Design
dotnet add package Bogus  # For generating fake data



# Export data to CSV
dotnet run --project InsureX.SeedTool export --type policies --format csv --from 2025-01-01 --to 2026-12-31

# Import data from CSV
dotnet run --project InsureX.SeedTool import --type policies --file policies_import.csv

# Generate test data
dotnet run --project InsureX.SeedTool generate --type all --count 100





dotnet build InsureX.slnx

dotnet build 2>&1 | grep -i error
